Tips: 该插件依赖 zepto.min.js，js压缩前4k，压缩后2k

  JS调用方法：

  1. 无链接的情况下：
  
  $('#sliderBox').touchslider({
	srcArr: [
	"http://fakeimg.pl/320x175/666666",
	"http://fakeimg.pl/320x175/ff0000/",
	"http://fakeimg.pl/320x175/222222/",
	"http://fakeimg.pl/320x175/999999/"
	]
  });
  
  2. 有链接的情况下：
  $('#sliderBox').touchslider({
    srcArr: [
  [
	"http://fakeimg.pl/320x175/666666",
	"http://www.hehaibao.com/"
  ],
  [
	"http://fakeimg.pl/320x175/ff0000",
	"http://www.hehaibao.com/"
  ],
  [
	"http://fakeimg.pl/320x175/222222",
	"http://www.hehaibao.com/"
  ],
  [
	"http://fakeimg.pl/320x175/999999",
	"http://www.hehaibao.com/"
  ]
  ]
  });
  
	
   其他自定义参数：

  $('#sliderBox').touchslider({
	box: '#sliderBox',
	srcArr: [],
	arrows: false,
	auto: true,
	autoTime: 2000
  });
